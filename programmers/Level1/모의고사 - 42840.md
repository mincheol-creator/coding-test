def solution(answers):
    answer = []
    person1=[1,2,3,4,5]
    person2=[2,1,2,3,2,4,2,5]
    person3=[3,3,1,1,2,2,4,4,5,5]
    result1=0
    result2=0
    result3=0
    temp=0

    for i in range(0,len(answers)):
        if answers[i]==person1[i%5]:
            result1+=1
        if answers[i]==person2[i%8]:
            result2+=1
        if answers[i]==person3[i%10]:
            result3+=1

    if result1>result2:
        answer.append(1)
        temp=result1
    elif result1==result2:
        answer.append(1)
        answer.append(2)
        temp=result1
    else:
        answer.append(2)
        temp=result2

    if temp>result3:
        print()
    elif temp==result3:
        answer.append(3)
    else:
        answer.clear()
        answer.append(3)

    return answer
