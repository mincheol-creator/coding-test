```python
def solution(a, b):
    answer = ''
    month=[31,29,31,30,31,30,31,31,30,31,30,31]
    day_sum=0
    for i in range(a-1):
        day_sum+=month[i]
    day_sum+=b
    day=day_sum%7
    if day==1 :
        answer+='FRI'
    elif day==2 :
        answer+='SAT'
    elif day==3 :
        answer+='SUN'
    elif day==4 :
        answer+='MON'
    elif day==5 :
        answer+='TUE'
    elif day==6 :
        answer+='WED'
    elif day==0 :
        answer+='THU'
    return answer
```

```javascript
function solution(a, b) {
  var answer = 0;
  var month = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
  var day = ["THU", "FRI", "SAT", "SUN", "MON", "TUE", "WED"];
  for (var i = 1; i < a; i++) {
    answer += month[i - 1];
  }
  return day[(answer + b) % 7];
}
```
