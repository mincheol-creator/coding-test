def solution(n, lost, reserve):
    answer = 0
    student_arr=[]
    j=0
    for i in range(0, n):
        student_arr.append([i,1])
    for i in reserve:
        student_arr[i-1][1]=2
    for i in lost:
        if student_arr[i-1][1]==2:
            student_arr[i-1][1]=1
        else:
            student_arr[i-1][1]=0
    for i in range(0,n):
        if student_arr[i][1]==0:
            if i==0:
                if student_arr[1][1]==2:
                    student_arr[0][1]=1
                    student_arr[1][1]=1
            elif i>=1 and i<n-1:
                for j in range(0,2):
                    if student_arr[i-1][1]==2:
                        student_arr[i-1][1]=1
                        student_arr[i][1]=1
                    elif student_arr[i+1][1]==2:
                        student_arr[i][1]=1
                        student_arr[i+1][1]=1
            elif i==n-1:
                if student_arr[i-1][1]==2:
                    student_arr[i-1][1]=1
                    student_arr[i][1]=1
    for i in range(n):
        if student_arr[i][1]==1 or student_arr[i][1]==2:
            answer+=1
    print(student_arr)
    return answer
