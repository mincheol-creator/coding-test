### 효율성 문제

##### Python

```python
def solution(n):
    answer = 0
    list_stack=[]
    result_stack=[]
    for i in range(2,n+1):
        for j in range(1,i+1):
            if i%j==0:
                list_stack.append(j)
        if len(list_stack)==2:
            result_stack.append(j)
        list_stack=[]
    answer = len(result_stack)
    print(answer)

    return answer
```

def solution(n):
    answer = 0
    sum=0
    result_stack=[]
    for i in range(2,n+1):
        for j in range(1,i+1):
            if i%j==0:
                sum+=j
        if i==(sum-1):
            result_stack.append(i)
        sum=0
    answer = len(result_stack)
    print(answer)

    return answer

def solution(n):
    answer = 0
    sum=0
    add=1
    result_stack=[]
    for i in range(2,n+1):
        while sum<=1:
            if i%add==0:
                sum+=add
            add+=1
        if i==(sum-1):
            result_stack.append(i)
        sum=0
        add=1
    answer = len(result_stack)
    print(answer)

    return answer
