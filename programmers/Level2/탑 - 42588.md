def solution(heights):
    answer = []
    for i in range(-1,-len(heights)-1,-1):
        for j in range(i-1,-len(heights)-1,-1):
            if j==-len(heights):
                if heights[i]<heights[j]:
                    answer.insert(0, j+len(heights)+1)
                    break
                else:
                    answer.insert(0, 0)
                    break
            if heights[i]<heights[j]:
                answer.insert(i, j+len(heights)+1)
                break
    answer.insert(0,0)


    return answer
