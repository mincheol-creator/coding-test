```python
def solution(number, k):
    arr=[]
    for i, num in enumerate(number):
        if i==0:
            arr.append(num)
        else:
            while len(arr)>0 and arr[-1]<num and k!=0:
                if arr[-1]<num:
                    arr.pop()
                    k-=1
        if k==0:
            arr+=list(number[i:])
            break
        if i!=0:
            arr.append(num)
    if k>0:
        for i in range(k):
            arr.pop()
    return "".join(i for i in arr)
```

효율성 문제

```python
def solution(number, k):
    arr=[]
    count=0
    subtract=0
    for i in number:
        arr.append(i)
    while k>0:
        for i in arr[count:k+count+1]:
            max_num=max(arr[count:k+count+1])
            if i==max_num:
                break
            else:
                subtract+=1
        for i in range(subtract):
            arr.pop(count)
        k=k-subtract
        count+=1
        subtract=0

    return "".join(i for i in arr)
```
