효율성 문제

```python
def solution(number, k):
    answer = ''
    arr=[]
    count=0
    for i in number:
        arr.append(i)
    for i in range(k):
        while True:
            try:
                if arr[count]<arr[count+1]:
                    arr.pop(count)
                    break
                if count==len(number)-1:
                    break
            except:
                arr.pop(count)
                break
            count+=1
        if count==0:
            count=0
        else:
            count-=1
    for i in arr:
        answer+=i
    return answer
```
